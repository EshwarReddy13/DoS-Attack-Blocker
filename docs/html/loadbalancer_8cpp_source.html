96void LoadBalancer::scaleServers() {
97 int load = request_queue.size();
98 int active = servers.size();
99 int busy_servers = 0;
100
101 // Count busy servers
102 for (auto* s : servers) {
103 if (s->isBusy()) busy_servers++;
104 }
105
106 // If overloaded, add a server (up to max)
107 if (load > active * 2 && active < max_servers) {
108 servers.push_back(new WebServer());
109 std::cout << " [SCALE UP] Added server. Total: " << servers.size() << std::endl;
110 }
111 // If underloaded and more than min servers, remove one (only if no busy servers)
112 // Use a more conservative threshold: scale down only when queue is very small
113 else if (load <= 5 && active > min_servers && busy_servers == 0) {
114 delete servers.back();
115 servers.pop_back();
116 std::cout << " [SCALE DOWN] Removed server. Total: " << servers.size() << std::endl;

138Request* LoadBalancer::generateRandomRequest(int arrival_time) {
139 std::string in = generateRandomIP();
140 std::string out = generateRandomIP();
141
142 // Debug: Check if IPs are valid
143 if (in.empty() || out.empty()) {
144 std::cerr << "Error: Generated empty IP address" << std::endl;
145 return nullptr;
146 }
147
148 // Check if source IP should be blocked
149 if (isIPBlocked(in)) {
150 blocked_requests++;
151 return nullptr; // Block the request
152 }
153
154 // Track request count for this IP (safer access)
155 if (ip_request_count.find(in) == ip_request_count.end()) {
156 ip_request_count[in] = 1;
157 } else {
158 ip_request_count[in]++;
159 }
160
161 // Check if this IP has made too many requests
162 if (ip_request_count[in] > max_requests_per_ip) {
163 blockIP(in);
164 blocked_requests++;
165 return nullptr; // Block the request
166 }
167
168 int proc_time = 1 + rand() % 10; // Range: 1-10 clock cycles
169 return new Request(in, out, proc_time, arrival_time);
170}

190bool LoadBalancer::isIPBlocked(const std::string& ip) {
191 // Check if IP is empty or too short
192 if (ip.empty() || ip.length() < 7) {
193 return false;
194 }
195
196 // Check if IP is in blocked set
197 if (blocked_ips.find(ip) != blocked_ips.end()) {
198 return true;
199 }
200
201 // Block certain IP ranges (simulating firewall rules)
202 // Block 192.168.x.x (private network)
203 if (ip.length() >= 8 && ip.substr(0, 8) == "192.168.") {
204 return true;
205 }
206
207 // Block 10.x.x.x (private network)
208 if (ip.length() >= 3 && ip.substr(0, 3) == "10.") {
209 return true;
210 }
211
212 // Block 127.x.x.x (localhost)
213 if (ip.length() >= 4 && ip.substr(0, 4) == "127.") {
214 return true;
215 }
216
217 return false;
